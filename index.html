Control Mapping
With up to four players sharing one keyboard, letting them remap keys to a comfortable layout was essential. A visual representation of the mapping was also needed. The below script shows how this was achieved.
key_remap.gd
extends Control
@export var key_shading: Dictionary[String,TextureRect]
@export var input_instruction1: Label
@export var input_key_display: Label
@export var input_instruction2: Label
@export var input_instruction3: Label
@export var keyboard: TextureRect
@export var confim_buttons: PanelContainer
@export var button_1: Button
var is_started: bool = false

# [action_name, readable label, rat_index]
const MAPPINGS: Array = [
    ["RAT1_forward",  "Rat 1 Forward",  0],
    ["RAT1_backward", "Rat 1 Backward", 0],
    ["RAT2_forward",  "Rat 2 Forward",  1],
    ["RAT2_backward", "Rat 2 Backward", 1],
    ["RAT3_forward",  "Rat 3 Forward",  2],
    ["RAT3_backward", "Rat 3 Backward", 2],
    ["RAT4_forward",  "Rat 4 Forward",  3],
    ["RAT4_backward", "Rat 4 Backward", 3],
]
var current_input_event: InputEvent
var assigned_keys: Dictionary = {} # action_name -> key text
var counter: int = 0

func begin() -> void:
    is_started = true
    prompt_next()

func _input(event: InputEvent) -> void:
    if is_started:
        if event is InputEventKey and event.pressed:
            print(event)
            # grab the input event as current_input_event
            current_input_event = event
            # show the plain-text character / key name
            input_key_display.text = event.as_text_key_label()
            submit_key()

# Prepares to take the next input
func prompt_next() -> void:
    input_key_display.text = ""
    if counter >= MAPPINGS.size():
        is_started = false
        input_instruction1.text = ""
        input_instruction3.text = ""
        show_confim()
        return
    var label: String = MAPPINGS[counter][1]
    input_instruction1.text = "Press the key that will move %s:" % label

func submit_key() -> void:
    # Error handling
    if input_key_display.text == "":
        input_instruction3.text = "You must select a key"
        return
    if input_key_display.text in assigned_keys.values():
        input_instruction3.text = "Key already mapped, select another"
        return
    if !key_shading.has(input_key_display.text.to_lower()):
        input_instruction3.text = "NAY! Pick another."
        return
    var action_name: String = MAPPINGS[counter][0]
    var label: String = MAPPINGS[counter][1]
    # applies the rat colour to the chosen key
    var color: Color = Globals.rat_colours[MAPPINGS[counter][2]]
    InputMap.action_add_event(action_name, current_input_event)
    assigned_keys[action_name] = input_key_display.text
    add_to_key_display(label, input_key_display.text, color)
    counter += 1
    input_instruction3.text = ""
    prompt_next()

# adds the key / colour to the on-screen keyboard display
func add_to_key_display(label: String, key_text: String, color: Color) -> void:
    var key = key_text.to_lower()
    key_shading[key].show()
    key_shading[key].modulate = color

func show_confim() -> void:
    confim_buttons.show()
    button_1.grab_focus()

func reset() -> void:
    confim_buttons.hide()
    counter = 0
    for key in key_shading.keys():
        key_shading[key].hide()
    assigned_keys.clear()
    is_started = true
    prompt_next()

func next_scene() -> void:
    get_parent().confirm()
Rat King Movement
Moving the player became complicated when 8 inputs were involved. Each of the 4 individual rats could be moved forwards and backwards, which was displayed visually by physically moving each rat sprite.
rat_movement.gd
var total_movement = Vector2.ZERO
var rats = [
    { "sprite": rat1, "prefix": "RAT1" },
    { "sprite": rat2, "prefix": "RAT2" },
    { "sprite": rat3, "prefix": "RAT3" },
    { "sprite": rat4, "prefix": "RAT4" },
]

for x in rats.size():
    rats[x].sprite.get_child(1).set_animation(&"default")

    var fwd: bool = Input.is_action_pressed(rats[x].prefix + "_forward")
    var back: bool = Input.is_action_pressed(rats[x].prefix + "_backward")
    if is_control_inverted:
        fwd = Input.is_action_pressed(rats[x].prefix + "_backward")
        back = Input.is_action_pressed(rats[x].prefix + "_forward")
    else:
        fwd = Input.is_action_pressed(rats[x].prefix + "_forward")
        back = Input.is_action_pressed(rats[x].prefix + "_backward")

    rat_input_changed.emit(rats[x].prefix, fwd, back)
    if rat_hunger[x] < get_parent().max_rat_hunger:
        var input_strength := 0.0
        if fwd:
            input_strength += 1.0
            rats[x].sprite.get_child(1).position = rat_sprite_position + Vector2(100, 0)
        if back:
            input_strength -= 1.0
            rats[x].sprite.get_child(1).position = rat_sprite_position + Vector2(-100, 0)
        if input_strength != 0.0:
            turn_on_rat(rats[x].sprite)
            rat_hunger[x] += 1 * delta
            rat_hunger_changed.emit(x, rat_hunger[x])
            var direction = Vector2.RIGHT.rotated(rats[x].sprite.global_rotation)
            total_movement += direction * input_strength
        else:
            rats[x].sprite.get_child(1).position = rat_sprite_position
            turn_off_rat(rats[x].sprite)
    else:
        rats[x].sprite.get_child(1).set_animation(&"tired")

velocity = total_movement * (Vector2(base_speed, base_speed) + Vector2(get_parent().score * 20, get_parent().score * 20)) * speed_mod

About
A “Papers, Please”-like set aboard a retro space station. You play the Post Master, directing post to the correct destination — if it meets the day’s ever-changing sending criteria. To preserve the retro feel, destinations and criteria appear on a functioning monochrome monitor, an exploration of SubViewports in a 3D environment. Players can open letters, read them and redact them, built from Control nodes with a clickable button for every word.
